\section{Formal Grammar Definition}

\begin{flushleft}

\item N = \{E\} \\

\begin{itemize}
\item R = \{ \\ 
\item \hspace{10mm} E $\rightarrow$ E + E \\
\item \hspace{10mm} E $\rightarrow$ E - E \\
\item \hspace{10mm} E $\rightarrow$ E * E \\
\item \hspace{10mm} E $\rightarrow$ E / E \\
\item \hspace{10mm} E $\rightarrow$ ( E ) \\
\item \hspace{10mm} E $\rightarrow$ \textbf{i} \\
\end{itemize}
\}
\end{itemize}
